\documentclass{pkgtest}
\testpackage{svn-multi}

\testnum{3}
\testtitle{Usage of \cs{svnkwsave}}

% Note that the $ at the end must have a leading space!
\svnkwsave{$Testkw1: test value $}
\svnkwsave{$Testkw2: test value with : colon  $}

\begin{document}
\svnkwsave{$Testkw3: test value after begin document $}
\maketitle

\section*{Keywords}
\noindent
Testkw1: "\svnkw{Testkw1}"\\
Testkw2: "\svnkw{Testkw2}"\\
Testkw3: "\svnkw{Testkw3}"\\

\testequal{test value}{\svnkw{Testkw1}}[Normal Keyword]
\testequal{test value with : colon}{\svnkw{Testkw2}}[Keyword with colon]
\testequal{test value after begin document}{\svnkw{Testkw3}}[Keyword after begin document]

\section*{Overwritten Keywords}
\svnkwsave{$Testkw1: overwritten value $}
Testkw1: ``\svnkw{Testkw1}''\\

\testequal{overwritten value}{\svnkw{Testkw1}}[Overwritten value]

\end{document}
